Dump vTable

On this page

NodeJS script I made to dump virtual function indexs from LeviLamina ​

Format:

[Index] functionName(params) -> returnType

Example: ClientInstance_vtable.txt

[1] onInitMinecraftGame() -> void
[15] requestLeaveGame(bool, bool) -> void

Adding vTables ​

You can specify what classes you want the vTables for by adding to the structs array, just add the path to the file with no line number #L14 at the end and without the starting part /src-server/mc/ so you should just be left with something like world/actor/Actor.h

js
import fs from "fs";
import https from "https";
import path from "path";

// Update to latest commit on levilamina
const LEVI_COMMIT = "4726f6d71b90a94f11c23023376028753d482bd0";

const URI = `https://raw.githubusercontent.com/LiteLDev/LeviLamina/${LEVI_COMMIT}/src-server/mc/`;

const structs = ["world/actor/Actor.h", "client/game/ClientInstance.h", "world/level/Level.h"];

for (const struct of structs) {
    dump(struct);
}

function dump(url) {
    url = URI + url;

    const fileName = path.basename(url);
    const outputFile = fileName.replace(/\.[^.]+$/, "") + "_vtable.txt";

    https.get(url, (res) => {
        if (res.statusCode !== 200) {
            console.error(`Failed to download file: ${res.statusCode} ${res.statusMessage}`);
            res.resume();
            process.exit(1);
        }

        let data = "";
        res.on("data", (chunk) => (data += chunk));
        res.on("end", () => {
            const pattern =
                /\/\/ vIndex:\s*(\d+)\s*\n\s*virtual\s+([\w:<>~\s*&]+?)\s+([A-Za-z_~][A-Za-z0-9_:]*)\s*\(([^)]*)\)/gs;

            const matches = [...data.matchAll(pattern)];

            const results = matches.map((m) => {
                const index = m[1];
                const returnType = m[2].replace(/\s+/g, " ").trim();
                const funcName = m[3];
                const params = m[4].replace(/\s+/g, " ").trim();
                return `[${index}] ${funcName}(${params}) -> ${returnType}`;
            });

            fs.writeFileSync(outputFile, results.join("\n"), "utf8");
            console.log(`Dumped ${results.length} functions into ${outputFile}`);
        });
    }).on("error", (err) => {
        console.error("Download error:", err.message);
    });
}
